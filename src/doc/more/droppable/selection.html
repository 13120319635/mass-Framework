 $.require("ready,more/droppable",function(){
                $("html").droppable({
                    dragstart:function(event,dd){
                        dd.dragger =  $('<div class="selection" />')
                        .css('opacity', .65 )
                        .appendTo( "bodu" );
                        dd.dragger.data("_mass_dd",dd)
                    },
                    drag:function( ev, dd ){
                        dd.dragger.css({
                            top: Math.min( ev.pageY, dd.startY ),
                            left: Math.min( ev.pageX, dd.startX ),
                            height: Math.abs( ev.pageY - dd.startY ),
                            width: Math.abs( ev.pageX - dd.startX )
                        });
                    },
                    dragend:function(event,dd){
                        dd.dragger.remove()
                    },
                    range:".drop",
                    tolerance: "intersect",
                    hoverClass:"active",
                    drop: function(event, dd){
                        dd.dropper.toggleClass("dropped");
                    }
                });
});
